# 开启jupyter服务使用以下指令
srun --nodelist=slurmd-5 --job-name=inference --gres=gpu:1 jupyter lab --ip 0.0.0.0 --port=30005

# 抢占式申请实验室机器时先调用下面这条指令，再将返回的job id替换到第二条指令中
salloc --partition=fnlp-x090  --job-name=bash --gres=gpu:4 --cpus-per-task=8
# salloc --partition=a800  --job-name=bash --gres=gpu:1 --cpus-per-task=8
srun --pty --jobid=<JOBID> bash
